# Intent Classification
class IntentAgent:
    def classify(self.message):
        text = message.lower()
        if "refund" in text:
            return "refund", "high"
            if "cancal" in text :
                return "cancellation", "high"
                if "invoice" in text or "bill" in text :
                    return "billing", "medium" 
                    if "text" in text : 
                        return  "general_help", "low"
                        return "general", "low"

# Reply genaration 
class ReplyAgent:
    def create_reply(self.message, intent,urgency):
        if intent  == "refund":
            return  "I understand you want a refund. Please share your order ID so I can assist you further."
        if intent == "cancellation" : 
            return "I can help you cancal your subscripton. Kindly provide your registered email."
            if intent  == "billing" : 
                return "It seems you have billing concern. Please send you invoice number of  verification."
                if intent == "general_help" :
                    return  "Sure, I'm here to help, could you please show more details?"
                    return "Thank you for your message. How can I assist you today?"

# Coordinator ties with 3 features together
class Coordinator:
    def __init__(self):
        self.intent_agent = IntentAgent()
        self.reply_agent = ReplyAgent()
        self.memory = Memory()
        def ask(self,message):
            self.memory.add("user", message)
            intent, urgency = self.intent_agent.classify(message)
            reply = self.reply_agent.create_reply(message, intent, urgency)

            final_output = {
                "intent" : intent,
                "urgency" : urgency,
                "reply" : reply
            }
            self.memory.add("agent", reply)
            return final_output
